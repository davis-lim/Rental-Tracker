**File 3: src/pages/PropertiesPage.jsx**

Main properties list page.

State:
- `properties` (array, default [])
- `loading` (boolean, default true)
- `formOpen` (boolean, default false)
- `editingProperty` (object | null, default null)
- `deleteTarget` (object | null, default null) — property being deleted
- `dependentCounts` ({ tenants: 0, mortgages: 0 })
- `apiError` (string | null)

Functions:
- `fetchProperties()` — GET /api/properties, set properties, set loading false. On error set apiError.
- `handleSave({ address, notes })` — if editingProperty, PUT /api/properties/:id; else POST /api/properties. On success: close dialog, fetchProperties(). On error: set apiError "Something went wrong. Please try again."
- `handleDeleteClick(property)` — GET /api/properties/:id/dependents, set dependentCounts, set deleteTarget to property, open alert dialog.
- `handleDeleteConfirm()` — DELETE /api/properties/:deleteTarget.id?confirm=true. On success: close dialog, fetchProperties(). On error: set apiError "Could not delete. Please try again."
- `handleEditClick(property)` — set editingProperty, open form dialog.
- `handleAddClick()` — set editingProperty to null, open form dialog.

useEffect on mount: fetchProperties().

Render:
```jsx
<div className="pt-8">
  {/* Page header: flex row with heading and Add button */}
  <div className="flex items-center justify-between mb-8">
    <h2 className="text-xl font-semibold">Properties</h2>
    <Button onClick={handleAddClick}>Add Property</Button>
  </div>

  {/* Error banner */}
  {apiError && (
    <div className="bg-destructive/10 text-destructive text-sm p-3 rounded mb-4">
      {apiError}
    </div>
  )}

  {/* Loading state */}
  {loading && <p className="text-muted-foreground">Loading...</p>}

  {/* Empty state (per Copywriting Contract) */}
  {!loading && properties.length === 0 && (
    <div className="text-center py-12">
      <h3 className="text-lg font-semibold mb-2">No properties yet</h3>
      <p className="text-muted-foreground">Add your first rental property to start tracking tenants and payments.</p>
    </div>
  )}

  {/* 2-column card grid (per D-01) */}
  {!loading && properties.length > 0 && (
    <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
      {properties.map(property => (
        <Card key={property.id} className="p-4">
          <div className="flex items-start justify-between">
            <div>
              {/* Link to property detail page (per D-05) */}
              <Link to={`/properties/${property.id}`} className="text-lg font-semibold hover:underline">
                {property.address}
              </Link>
              {/* Tenant count badge (per D-02) */}
              <div className="mt-2">
                <Badge variant="secondary">
                  {property.tenant_count} {property.tenant_count === 1 ? 'tenant' : 'tenants'}
                </Badge>
              </div>
            </div>
            {/* Action buttons (per D-02) */}
            <div className="flex gap-2">
              <Button variant="ghost" size="icon" aria-label="Edit property" onClick={() => handleEditClick(property)}>
                <Pencil className="h-4 w-4" />
              </Button>
              <Button variant="ghost" size="icon" aria-label="Delete property" onClick={() => handleDeleteClick(property)}>
                <Trash2 className="h-4 w-4" />
              </Button>
            </div>
          </div>
        </Card>
      ))}
    </div>
  )}

  {/* Add/Edit Dialog */}
  <PropertyFormDialog
    open={formOpen}
    onOpenChange={(open) => { setFormOpen(open); if (!open) setEditingProperty(null); }}
    property={editingProperty}
    onSave={handleSave}
  />

  {/* Delete Confirmation (per UI-SPEC Delete Flow) */}
  <DeleteConfirmDialog
    open={!!deleteTarget}
    onOpenChange={(open) => { if (!open) setDeleteTarget(null); }}
    title="Delete property?"
    description={buildDeleteDescription()}
    cancelLabel="Keep property"
    confirmLabel="Delete property"
    onConfirm={handleDeleteConfirm}
  />
</div>
```

import { useState, useEffect } from 'react';
import { useParams, Link } from 'react-router-dom';
import { Button } from '@/components/ui/button';
import { Separator } from '@/components/ui/separator';

export default function PropertyDetailPage() {
  const { id } = useParams();
  const [property, setProperty] = useState(null);

  useEffect(() => {
    fetch(`/api/properties/${id}`)
      .then(r => r.json())
      .then(setProperty)
      .catch(() => {});

  return (

      <Link to="/properties" className="text-sm text-muted-foreground hover:text-foreground">&larr; Back to Properties</Link>
      <h2 className="text-xl font-semibold mt-4">{property.address}</h2>
      {property.notes && <p className="text-muted-foreground mt-1">{property.notes}</p>}
      <Separator className="my-6" />
      <h3 className="text-lg font-semibold">Tenants</h3>
      <p className="text-muted-foreground mt-2">Tenant management will be added in the next step.</p>
    </div>
  );
}
