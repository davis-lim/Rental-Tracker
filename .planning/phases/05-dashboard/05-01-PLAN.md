From server/services/rentPayments.js — computeDueDate helper (duplicate this exact logic):
```js
function computeDueDate(deadlineDay, month) {
  const [y, m] = month.split('-').map(Number);
  const daysInMonth = new Date(y, m, 0).getDate();
  const clampedDay = Math.min(deadlineDay, daysInMonth);
  return `${month}-${String(clampedDay).padStart(2, '0')}`;
}
```

Create server/services/dashboard.js with three exported functions. Use sync better-sqlite3 (import db from '../db.js'). Copy computeDueDate verbatim from rentPayments.js — do NOT import it cross-service.

**getSummary(month)**
- month param is YYYY-MM string (caller passes current month)
- Fetch all tenants with JOIN to properties: id, name, property_address, rent_amount, deadline_day, grace_period_days
- LEFT JOIN rent_payments ON tenant_id = tenants.id AND month = ?
- For each row: compute due_date via computeDueDate(deadline_day, month); derive status:
  - If paid_date is non-null: status = is_on_time === 1 ? 'paid_on_time' : 'paid_late'
  - Else: status = 'unpaid'
- Return array of: { type: 'rent', tenant_id, name, property_address, rent_amount, due_date, status, paid_date, amount_paid }
- Do same for mortgages: LEFT JOIN mortgage_payments ON mortgage_id = mortgages.id AND month = ?
- For each mortgage row: due_date = computeDueDate(due_day, month); same status logic (no grace period for mortgages — is_on_time already stored)
- Return array of: { type: 'mortgage', mortgage_id, lender, property_address, amount, due_date, status, paid_date, amount_paid }
- Return { tenants: [...], mortgages: [...] }

**getUpcoming(month, today)**
- today is YYYY-MM-DD string; window end = today + 7 days computed via:
  ```js
  const d = new Date(today + 'T00:00:00');
  d.setDate(d.getDate() + 7);
  const windowEnd = d.toISOString().slice(0, 10);
  ```
- Fetch all tenants; for each: dueDate = computeDueDate(deadline_day, month)
  - Check if no rent_payment exists for tenant+month (getOne pattern: SELECT * FROM rent_payments WHERE tenant_id=? AND month=?)
  - If dueDate >= today AND dueDate <= windowEnd AND unpaid → include as { type: 'rent', name, property_address, due_date: dueDate, amount: rent_amount }
- Fetch all mortgages; for each: dueDate = computeDueDate(due_day, month)
  - Check if no mortgage_payment for mortgage+month
  - If dueDate >= today AND dueDate <= windowEnd AND unpaid → include as { type: 'mortgage', lender, property_address, due_date: dueDate, amount }
- Sort combined result by due_date ASC
- Return array of items

**getOverdue(month, today)**
- For each tenant: compute dueDate = computeDueDate(deadline_day, month); add grace_period_days:
  ```js
  const d = new Date(dueDate + 'T00:00:00');
  d.setDate(d.getDate() + t.grace_period_days);
  const graceDeadline = d.toISOString().slice(0, 10);
  ```
  If graceDeadline < today AND no rent_payment for tenant+month → include as { type: 'rent', name, property_address, due_date: dueDate, grace_deadline: graceDeadline, amount: rent_amount }
- For each mortgage: dueDate = computeDueDate(due_day, month); no grace period
  If dueDate < today AND no mortgage_payment for mortgage+month → include as { type: 'mortgage', lender, property_address, due_date: dueDate, amount }
- Sort by due_date ASC
- Return array of items

Export: export { getSummary, getUpcoming, getOverdue }; export default { getSummary, getUpcoming, getOverdue };

<task type="auto">
  <name>Task 2: Dashboard routes + mount</name>
  <files>server/routes/dashboard.js, server/routes.js</files>
  <action>
Create server/routes/dashboard.js:

```js
import { Router } from 'express';
import { getSummary, getUpcoming, getOverdue } from '../services/dashboard.js';

const router = Router();

// GET /api/dashboard/summary?month=YYYY-MM
router.get('/summary', (req, res) => {
  const month = req.query.month || new Date().toISOString().slice(0, 7);
  try {
    res.json(getSummary(month));
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// GET /api/dashboard/upcoming?month=YYYY-MM&today=YYYY-MM-DD
router.get('/upcoming', (req, res) => {
  const month = req.query.month || new Date().toISOString().slice(0, 7);
  const today = req.query.today || new Date().toISOString().slice(0, 10);
  try {
    res.json(getUpcoming(month, today));
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

// GET /api/dashboard/overdue?month=YYYY-MM&today=YYYY-MM-DD
router.get('/overdue', (req, res) => {
  const month = req.query.month || new Date().toISOString().slice(0, 7);
  const today = req.query.today || new Date().toISOString().slice(0, 10);
  try {
    res.json(getOverdue(month, today));
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

export default router;
```

Then edit server/routes.js: add import for dashboardRouter and mount it:
